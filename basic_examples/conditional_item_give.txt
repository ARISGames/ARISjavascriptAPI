<script type='text/javascript'>
var ARIS = {};
ARIS.ready = function()
{
  var item_name = 'SimpleItem';
  var item_id = ARIS.cache.idForItemName(item_name);
  var current_qty = ARIS.cache.getPlayerItemCount(item_id);
  if(current_qty < 5)
  {
    ARIS.givePlayerItemCount(item_id,1);
  }
  else
  {
    ARIS.takePlayerItemCount(item_id,1);
  }
}
</script>
